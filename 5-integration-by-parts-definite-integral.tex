\part[4]
Let $c$ be a real number. Use integration by parts to show that
\[
    \int x\sin(2x) \dif x = \frac{1}{4} \sin(2x) - \frac{1}{2} x\cos(2x) + c.
\]

\begin{EnvFullwidth}
\begin{solutionorgrid}[4in]
Put $u = x$ and $\dif v = \sin(2x) \dif x$. By the DI method

\vspace{.125in}

\begin{center}
	\begin{tabular}{lcc}

	\addlinespace
	$-$ & $1$\tikzmark{q} & \tikzmark{r}$-\frac{1}{2}\cos(2x)$ \\
	\addlinespace
	$+$ & $0$\tikzmark{s} & \tikzmark{t}$-\frac{1}{4}\sin(2x)$ \\
	\addlinespace
	\bottomrule
	\end{tabular}
	\begin{tikzpicture}[

\[
    \int x\sin(2x) \dif x = \frac{1}{4}\sin(2x) - \frac{1}{2}x\cos(2x) + c.

\part[4]
Show that the area of one of the regions is three times the area of the other region. 

\begin{EnvFullwidth}
\begin{solutionorgrid}[3.5in]
The left region has
\begin{align*}
    A_L &= \int_0^{\sfrac{\pi}{2}} x\sin(2x) \dif x \\
    &= \Eval{\frac{1}{4} \sin(2x) - \frac{1}{2} x\cos(2x)}{0}{\sfrac{\pi}{2}} \\
    &= -\frac{1}{2} \times \frac{\pi}{2} \times -1 - 0 \\
    &= \frac{\pi}{4}.
\end{align*}
The right region has
\begin{align*}
    A_R &= \int_{\sfrac{\pi}{2}}^{\pi} x\sin(2x) \dif x \\
    &= \Eval{\frac{1}{4} \sin(2x) - \frac{1}{2} x\cos(2x)}{\sfrac{\pi}{2}}{\pi} \\
    &= -\frac{1}{2} \times \pi \times 1 - \left( -\frac{1}{2} \times \frac{\pi}{2} \times -1 \right) \\
    &= -\frac{3\pi}{4}.
\end{align*}
\end{solutionorgrid}
\end{EnvFullwidth}
