\begin{parts}

\part[2]
Find, to four decimal places, the value of $\displaystyle{\int_{\sfrac{\pi}{6}}^{\sfrac{\pi}{4}} \cot(\theta) \dif \theta}$.

\begin{EnvFullwidth}
\begin{solutionorgrid}[.75in]
By technology
\[
    \int_{\sfrac{\pi}{6}}^{\sfrac{\pi}{4}} \cot(\theta) \dif \theta \approx 0.3466.
\]
\end{solutionorgrid}
\end{EnvFullwidth}

\part[4]
Show that $\displaystyle{\int_{\sfrac{\pi}{6}}^{\sfrac{\pi}{4}} \cot(\theta) \dif \theta = \ln(\sqrt{2})}$. \emph{Hint}: $\displaystyle{\cot(\theta) = \frac{\cos(\theta)}{\sin(\theta)}}$.

\begin{EnvFullwidth}
\begin{solutionorgrid}[5in]
Let $u = \sin(\theta)$, so $\dif u = \cos(\theta) \dif \theta$ and $\displaystyle{\dif \theta = \frac{\dif u}{\cos(\theta)}}$. At the limits
\[
    \theta = \frac{\pi}{6}, \qquad u = \frac{1}{2}, \qquad \theta = \frac{\pi}{4} , \qquad u = \frac{1}{\sqrt{2}}.
\]
On $u$-substitution we get
\begin{align*}
    \int_{\sfrac{\pi}{6}}^{\sfrac{\pi}{4}} \frac{\cos(\theta)}{\sin(\theta)} \dif \theta &= \int_{\sfrac{1}{2}}^{\sfrac{1}{\sqrt{2}}} \frac{\cancel{\cos(\theta)}}{u} \frac{\dif u}{\cancel{\cos(\theta)}} \\
    &= \Eval{\ln(u)}{\sfrac{1}{2}}{\sfrac{1}{\sqrt{2}}} && (u > 0) \\
    &= \ln(2^{-\sfrac{1}{2}}) - \ln(2^{-1}) \\
    &= \ln(2^{\sfrac{1}{2}}) \\
    &= \ln(\sqrt{2}).
\end{align*}
\end{solutionorgrid}
\end{EnvFullwidth}
