\question % 2018 Exam, Q6.
Let $t$ be the time, in seconds. An object has a velocity, in metres per second, given by
\[
    v(t) = \E^{\cos^2(t)} \sin(2t).
\]

\part[3]
Find the \emph{exact} distance travelled by the object between $t = 0$ and $\displaystyle{t = \frac{\pi}{2}}$ seconds.

\begin{EnvFullwidth}
\begin{solutionorgrid}[2.5in]
Let $u = \cos^2(t)$, so $\dif u = -2\sin(t)\cos(t) \dif t$ and $\displaystyle{\dif t = \frac{\dif u}{-\sin(2t)}}$. At the limits
\[
    t = 0, \qquad u = 1, \qquad t = \frac{\pi}{2}, \qquad u = 0.
\]
On $u$-substitution
\begin{align*}
    d &= \int_0^{\sfrac{\pi}{2}} \E^{\cos^2(t)} \sin(2t) \dif t \\
    &= \int_1^0 \E^u \cancel{\sin(2t)} \frac{\dif u}{-\cancel{\sin(2t)}} \\
    &= \int_0^1 \E^u \dif u \\
    &= \Eval{\E^u}{0}{1} \\
    &= \E - 1 \,\, \si{\meter}.
\end{align*}
\end{solutionorgrid}
\end{EnvFullwidth}

\part[1]
Hence, find the distance travelled between $t = 0$ and $t = \pi$ seconds.

\begin{EnvFullwidth}
\begin{solutionorgrid}[1.5in]
On the basis of symmetry the total is
\[
    D = 2(\E  - 1) \,\, \si{\meter}.
\]
\end{solutionorgrid}
\end{EnvFullwidth}
