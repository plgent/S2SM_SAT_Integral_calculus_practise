\part[3]
Let $c$ be a real number. Using the fact that $\sin^3(x) = \sin^2(x)\sin(x)$, show that
\[
    \int \sin^3(x) \dif x = -\cos(x) + \frac{1}{3}\cos^3(x) + c.
\]

\begin{EnvFullwidth}
\begin{solutionorgrid}[2in]
Since $\sin^3(x) = (1 - \cos^2(x))\sin(x)$, on putting $u = \cos(x)$ there is
\[
    \dif u = -\sin(x)\dif x, \qquad \dif x = \frac{\dif u}{-\sin(x)}.
\]
Thus,
\begin{align*}
    \int \sin^3(x) \dif x &= \int (1 - u^2)\cancel{\sin(x)} \frac{\dif u}{-\cancel{\sin(x)}} \\
    &= -\int 1 - u^2 \dif u \\
    &= -(u - \frac{1}{3} u^3 + c_1) \\
    &= -\cos(x) + \frac{1}{3}\cos^3(x) + c.
\end{align*}
\end{solutionorgrid}
\end{EnvFullwidth}

\subpart[1]
Find $f(\frac{\pi}{2})$.

\begin{EnvFullwidth}
\begin{solutionorgrid}[.75in]
We have
\[
    f(\tfrac{\pi}{2}) = 1.

\subpart[1]
Explain why the function $f(x)$ has an inverse function.

\begin{EnvFullwidth}
\begin{solutionorgrid}[.75in]
The inverse exists because $f$ is injective (one-to-one).
\end{solutionorgrid}
\end{EnvFullwidth}

\part[4]
Use parts (a) and (b) to show that $\displaystyle{\int_0^1 f^{-1}(x) \dif x = \frac{\pi}{2} - \frac{2}{3}}$.

\begin{EnvFullwidth}
\begin{solutionorgrid}[4.5in]
By symmetry, the area under the curve is the area of the rectangle of base $1$ and height $\frac{\pi}{2}$ minus the area under the original function. Thus,
\begin{align*}
    \int_0^1 f^{-1}(x) \dif x &= \frac{\pi}{2} \times 1 - \int_0^{\sfrac{\pi}{2}} \sin^3(x) \dif x \\
    &= \frac{\pi}{2} + \Eval{\cos(x) - \frac{1}{3}\cos^3(x)}{0}{\sfrac{\pi}{2}} && (\textrm{by (a)}) \\
    &= \frac{\pi}{2} + \left(\cos(\tfrac{\pi}{2}) - \frac{1}{3}\cos^3(\tfrac{\pi}{2})\right) - \left(\cos(0) - \frac{1}{3}\cos^3(0)\right) \\
    &= \frac{\pi}{2} - \left(1 - \frac{1}{3}\right) \\
    &= \frac{\pi}{2} - \frac{2}{3}.
\end{align*}
\end{solutionorgrid}
\end{EnvFullwidth}
